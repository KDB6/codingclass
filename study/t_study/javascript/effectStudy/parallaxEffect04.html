script

재귀 함수를 사용하여 html에서 지정해준 content__item를 해당 브라우저 높이에 1/2가 되었을 떄 나타나도록 한다.

scrollTop은 개별적으로 사용하는 방법이 있지만, || 를 이용해 여러가지를 한번에 사용할 수도 있다.

코드 보기 
    // 재귀함수
    function scroll() {
        let scrollTop = window.pageXOffset || document.documentElement.scrollTop || window.scrollY
        
        document.querySelectorAll(".content__item").forEach(item => {
            if(scrollTop > item.offsetTop - window.innerHeight/2) {
                item.classList.add("show");
            }
        });

        requestAnimationFrame(scroll);
    }
    scroll();
