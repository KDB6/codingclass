01. 변수 : 데이터 불러오기

데이터 저장 가능하며. 불러오기 가능

        y = 200,
        z = "javascript";

document.write(x);

document.write(z);

100
200

document.write(arr[0]);
document.write(arr[1]);
document.write(arr[2]);
document.write(arr[3]);
document.write(arr[4]);

    document.write(arr[i]);
};

07. 배열 : 데이터 불러오기 : forEach()

배열 데이터 불러오기 다섯번째 방법

const num = [100, 200, 300, 400, 500];

num.forEach(function(el) {
    document.write(el);

num.forEach(function(el, i, array) {     //el은element, i는index
    document.write(el);
    document.write(i);
    document.write(array);
});
결과보기
100
200
300
400
500

    document.write(el);
});
