function solution(s) {
    let answer;
    let arr = [...s]
    if(arr.length % 3 == 0) {
        answer = arr.splice(arr.length/3-1, 2)
    } else {
        answer = arr.splice(Math.ceil(arr.length/3), 5)
    }
    return answer.join("").toUpperCase();
}

function solution(abc1, abc2) {

    let answer = [];
    abc1.reverse();
    abc2.reverse();
    let p1 = p2 = 2;

    while(p1 < abc1.length && p2 < abc2.length) {
        if(abc1[p1] !== abc2[p2]) {
            answer.push(abc1[p1++]);
            p2++
        }
        else if(abc1[p1] >= abc2[p2]) p1++;
        else p2++;
    }
    return answer;
}
