30. 클래스 상속

    class study {
        constructor(num, name, job) {
            this.num = num;
            this.name = name;
            this.job = job;
        }
            result() {
                document.write(this.num + ".내이름은" + this.name + "이며, 직업은" + this.job + "입니다.")
            }
        }

        class study2 extends study {
            constructor(num, name, job, age) {
                super(num, name, job);
                this.age = age;
            }
            result2() {
                document.write(this.num + ".내이름은" + this.name + "이며, 직업은" + this.job + "이며" + this.age + "살입니다.")
            }
        }

    const info1 = new study("1", "k", "a")
    const info2 = new study2("2", "k", "b", 100)

    info1.result();
    info2.result();
    info2.result2();
결과보기
1.내이름은k이며, 직업은a입니다.
2.내이름은k이며, 직업은b입니다.
2.내이름은k이며, 직업은b이며100살입니다.
